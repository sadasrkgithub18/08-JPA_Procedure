CREATE OR REPLACE NONEDITIONABLE PROCEDURE EMP_EXPERIENCE(ENO IN NUMBER, NAME OUT VARCHAR2, EXPERIENCE OUT NUMBER)
IS
        TEMP1 DATE;
        TEMP2 DATE;
BEGIN
        SELECT ENAME INTO NAME FROM EMP WHERE EMPNO = ENO;
        SELECT HIREDATE INTO TEMP1 FROM EMP WHERE EMPNO = ENO;
        SELECT SYSDATE INTO TEMP2 FROM DUAL;
        EXPERIENCE := (TEMP2 - TEMP1) / 365;
END;
